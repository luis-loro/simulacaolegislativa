// ============================================= // SIMULAÇÃO LEGISLATIVA — Apps Script // ============================================= var SS = SpreadsheetApp.getActiveSpreadsheet(); // ── Utilitários ──────────────────────────── function getSheet(name) { return SS.getSheetByName(name); } function sheetData(name) { var sh = getSheet(name); var rows = sh.getDataRange().getValues(); var headers = rows[0]; return rows.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function erro(msg) { return json({ ok: false, erro: msg }); } // ── Autenticação ─────────────────────────── function autenticar(user_id, pin) { var participantes = sheetData('participantes'); var p = participantes.find(function(r) { return String(r.id) === String(user_id) && String(r.pin) === String(pin); }); return p || null; } // ── Roteador principal ───────────────────── function doPost(e) { try { var params = JSON.parse(e.postData.contents); var action = params.action; if (action === 'login') return acao_login(params); if (action === 'buscar_votacoes') return acao_buscar_votacoes(params); if (action === 'registrar_voto') return acao_registrar_voto(params); if (action === 'buscar_meu_voto') return acao_buscar_meu_voto(params); if (action === 'buscar_resultado') return acao_buscar_resultado(params); if (action === 'buscar_progresso') return acao_buscar_progresso(params); if (action === 'admin_abrir') return acao_admin_abrir(params); if (action === 'admin_encerrar') return acao_admin_encerrar(params); if (action === 'admin_reabrir') return acao_admin_reabrir(params); if (action === 'admin_anular') return acao_admin_anular(params); return erro('Ação desconhecida: ' + action); } catch(err) { return erro('Erro interno: ' + err.message); } } // ── Ações ────────────────────────────────── function acao_login(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); return json({ ok: true, id: participante.id, nome: participante.nome, partido: participante.partido, admin: participante.admin === true || String(participante.admin).toLowerCase() === 'true' bancada: participante.bancada }); } function acao_buscar_votacoes(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); var votacoes = sheetData('votacoes'); return json({ ok: true, votacoes: votacoes }); } function acao_registrar_voto(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); // Verifica se a votação está aberta var votacoes = sheetData('votacoes'); var votacao = votacoes.find(function(v) { return String(v.id) === String(p.votacao_id); }); if (!votacao) return erro('Votação não encontrada'); if (votacao.status !== 'aberta') return erro('Votação não está aberta'); // Valida o voto var votosValidos = ['sim', 'nao', 'abstencao']; if (votosValidos.indexOf(p.voto) === -1) return erro('Voto inválido'); var sh = getSheet('votos'); var linhas = sh.getDataRange().getValues(); var agora = new Date().toISOString(); // Atualiza voto existente for (var i = 1; i < linhas.length; i++) { if (String(linhas[i][0]) === String(p.user_id) && String(linhas[i][1]) === String(p.votacao_id)) { sh.getRange(i + 1, 3).setValue(p.voto); sh.getRange(i + 1, 4).setValue(agora); return json({ ok: true, acao: 'atualizado' }); } } // Insere novo voto sh.appendRow([p.user_id, p.votacao_id, p.voto, agora]); return json({ ok: true, acao: 'inserido' }); } function acao_buscar_meu_voto(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); var votos = sheetData('votos'); var meuVoto = votos.find(function(v) { return String(v.user_id) === String(p.user_id) && String(v.votacao_id) === String(p.votacao_id); }); return json({ ok: true, voto: meuVoto ? meuVoto.voto : null }); } function acao_buscar_progresso(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); var votos = sheetData('votos'); var participantes = sheetData('participantes'); var votaram = votos.filter(function(v) { return String(v.votacao_id) === String(p.votacao_id); }).length; return json({ ok: true, votaram: votaram, total: participantes.length }); } function acao_buscar_resultado(p) { var participante = autenticar(p.user_id, p.pin); if (!participante) return erro('PIN inválido'); // Verifica se a votação está encerrada (ou anulada) var votacoes = sheetData('votacoes'); var votacao = votacoes.find(function(v) { return String(v.id) === String(p.votacao_id); }); if (!votacao) return erro('Votação não encontrada'); if (votacao.status !== 'encerrada' && votacao.status !== 'anulada') { return erro('Resultado ainda não disponível'); } var votos = sheetData('votos'); var participantes = sheetData('participantes'); var doJogo = votos.filter(function(v) { return String(v.votacao_id) === String(p.votacao_id); }); // Contagem geral var contagem = { sim: 0, nao: 0, abstencao: 0 }; doJogo.forEach(function(v) { if (contagem[v.voto] !== undefined) contagem[v.voto]++; }); // Detalhe por participante var detalhe = participantes.map(function(part) { var voto = doJogo.find(function(v) { return String(v.user_id) === String(part.id); }); return { id: part.id, nome: part.nome, partido: part.partido, voto: voto ? voto.voto : null }; }); return json({ ok: true, status: votacao.status, contagem: contagem, detalhe: detalhe }); } // ── Ações Admin ──────────────────────────── function verificarAdmin(user_id, pin) { var participante = autenticar(user_id, pin); if (!participante) return null; if (participante.admin !== true && participante.admin !== 'true') return null; return participante; } function atualizarStatusVotacao(votacao_id, novoStatus) { var sh = getSheet('votacoes'); var linhas = sh.getDataRange().getValues(); for (var i = 1; i < linhas.length; i++) { if (String(linhas[i][0]) === String(votacao_id)) { sh.getRange(i + 1, 3).setValue(novoStatus); return true; } } return false; } function acao_admin_abrir(p) { if (!verificarAdmin(p.user_id, p.pin)) return erro('Acesso negado'); var ok = atualizarStatusVotacao(p.votacao_id, 'aberta'); if (!ok) return erro('Votação não encontrada'); return json({ ok: true }); } function acao_admin_encerrar(p) { if (!verificarAdmin(p.user_id, p.pin)) return erro('Acesso negado'); var ok = atualizarStatusVotacao(p.votacao_id, 'encerrada'); if (!ok) return erro('Votação não encontrada'); return json({ ok: true }); } function acao_admin_reabrir(p) { if (!verificarAdmin(p.user_id, p.pin)) return erro('Acesso negado'); var ok = atualizarStatusVotacao(p.votacao_id, 'aberta'); if (!ok) return erro('Votação não encontrada'); return json({ ok: true }); } function acao_admin_anular(p) { if (!verificarAdmin(p.user_id, p.pin)) return erro('Acesso negado'); var ok = atualizarStatusVotacao(p.votacao_id, 'anulada'); if (!ok) return erro('Votação não encontrada'); return json({ ok: true }); }
